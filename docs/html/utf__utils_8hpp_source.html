177status_e utf::conversion::utf16_to_utf8(const std::basic_string_view<char16_t>& utf16_sv, std::basic_string<char8_t>& utf8_s, bool comply_with_standard = false) {
178 // Convert to UTF-32
179 std::basic_string<char32_t> code_points;
180 status_e status = utf16_to_utf32(utf16_sv, code_points, comply_with_standard);
181 if (status < status_e::success) {
182 return status;
183 }
184 // Convert to UTF-8
185 std::basic_string<char8_t> result_string;
186 status = utf32_to_utf8(code_points, result_string, comply_with_standard);
187 if (status < status_e::success) {
188 return status;
189 }
190
191 utf8_s = result_string;
192 return status_e::success;
193}

195status_e utf::conversion::utf32_to_utf8(const std::basic_string_view<char32_t>& utf32_sv, std::basic_string<char8_t>& utf8_s, bool comply_with_standard = false) {
196 std::basic_string<char8_t> result;
197
198 for (auto it = utf32_sv.begin(); it < utf32_sv.end(); it++) {
199 char32_t this_code_point = *it;
200 if (this_code_point > four_byte_boundary) {
201 return status_e::undefined_error;
202 }
203 if (this_code_point <= one_byte_boundary) {
204 result.push_back(this_code_point);
205 continue;
206 }
207 if (this_code_point <= two_byte_boundary) {
208 uint16_t cp_16 = static_cast<uint16_t>(this_code_point);
209
210 uint8_t first_5_bits = cp_16 >> 6;
211 uint8_t last_6_bits = (cp_16 << 10) >> 10;
212
213 result.push_back((double_byte_marker << 5) + first_5_bits);
214 result.push_back((trailing_byte_marker << 6) + last_6_bits);
215
216 continue;
217 }
218 if (this_code_point <= three_byte_boundary) {
219 uint16_t cp_16 = static_cast<uint16_t>(this_code_point);
220
221 bool wrong_encoding = comply_with_standard &&
222 cp_16 < low_surrogate_start &&
223 cp_16 > high_surrogate_start;
224 if (wrong_encoding) {
225 return status_e::non_standard_encoding;
226 }
227
228 uint8_t first_4_bits = cp_16 >> 12;
229 uint8_t middle_6_bits = (cp_16 << 4) >> 10;
230 uint8_t last_6_bits = (cp_16 << 10) >> 10;
231
232 result.push_back((triple_byte_marker << 4) + first_4_bits);
233 result.push_back((trailing_byte_marker << 6) + middle_6_bits);
234 result.push_back((trailing_byte_marker << 6) + last_6_bits);
235
236 continue;
237 }
238 if (this_code_point <= four_byte_boundary) {
239 uint8_t first_3_bits = this_code_point >> 18;
240 uint8_t second_6_bits = (this_code_point << 14) >> 26;
241 uint8_t third_6_bits = (this_code_point << 20) >> 26;
242 uint8_t last_6_bits = (this_code_point << 26) >> 26;
243
244 result.push_back((quadruple_byte_marker << 5) + first_3_bits);
245 result.push_back((trailing_byte_marker << 6) + second_6_bits);
246 result.push_back((trailing_byte_marker << 6) + third_6_bits);
247 result.push_back((trailing_byte_marker << 6) + last_6_bits);
248
249 continue;
250 }
251 }
252
253 return status_e::success;
254}

256status_e utf::conversion::utf16_to_utf32(const std::basic_string_view<char16_t>& utf16_sv, std::basic_string<char32_t>& utf32_s, bool comply_with_standard = false) {
257 // vector of UTF-32 characters (code points)
258 std::basic_string<char32_t> code_points;
259
260 bool was_double_character = false;
261 for (auto character_it = utf16_sv.begin(); character_it < utf16_sv.end(); character_it++) {
262 // skip this character, because we handled it last iteration
263 if (was_double_character) {
264 was_double_character = false;
265 continue;
266 }
267 // get this character
268 char16_t this_character = *character_it;
269
270 // check if this character is the last one
271 const bool exists_next_character = (character_it + 1) != utf16_sv.end();
272 char16_t next_character;
273
274 // get next character if there is one
275 if (exists_next_character) {
276 next_character = *(character_it + 1);
277 }
278
279 // if can be part of double character
280 if (is_high_surrogate(this_character)) {
281 // if there is no next character we add this as code point
282 if (!exists_next_character) {
283 if (!comply_with_standard) {
284 code_points.push_back(this_character);
285 continue;
286 }
287 return status_e::non_standard_encoding;
288 }
289
290 // if next character is not part of the double character we add this as code point
291 if (!is_low_surrogate(next_character)) {
292 if (!comply_with_standard) {
293 code_points.push_back(this_character);
294 continue;
295 }
296 return status_e::non_standard_encoding;
297 }
298
299 // it turned out we have a double character
300 was_double_character = true;
301
302 // do decoding "double UTF-16" -> UTF-32:
303 // https://en.wikipedia.org/wiki/UTF-16#Code_points_from_U+010000_to_U+10FFFF
304 char32_t high_code_point = (this_character - high_surrogate_start) << 10;
305 char16_t low_code_point = next_character - low_surrogate_start;
306 char32_t code_point = high_code_point + low_code_point + supplementary_plane_offset;
307 code_points.push_back(code_point);
308
309 continue;
310 }
311
312 // if not a double character add this as next code point
313 code_points.push_back(this_character);
314 }
315 // return resulting string
316 utf32_s = code_points;
317 return status_e::success;
318}
