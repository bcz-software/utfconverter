149 // https://en.wikipedia.org/wiki/UTF-16#Code_points_from_U+010000_to_U+10FFFF
150 // checks if UTF-16 character can be first in double character pair

181utf::conversion::status_e utf::conversion::utf16_to_utf32(const std::basic_string_view<char16_t>& utf16_sv, std::basic_string<char32_t>& utf32_s, bool comply_with_standard = false) {
182 // vector of UTF-32 characters (code points)
183 std::basic_string<char32_t> code_points;
184
185 bool was_double_character = false;
186 for (auto character_it = utf16_sv.begin(); character_it < utf16_sv.end(); character_it++) {
187 // skip this character, because we handled it last iteration
188 if (was_double_character) {
189 was_double_character = false;
190 continue;
191 }
192 // get this character
193 char16_t this_character = *character_it;
194
195 // check if this character is the last one
196 const bool exists_next_character = (character_it + 1) != utf16_sv.end();
197 char16_t next_character;
198
199 // get next character if there is one
200 if (exists_next_character) {
201 next_character = *(character_it + 1);
202 }
203
204 // if can be part of double character
205 if (is_high_surrogate(this_character)) {
206 // if there is no next character we add this as code point
207 if (!exists_next_character) {
208 if (!comply_with_standard) {
209 code_points.push_back(this_character);
210 continue;
211 }
212 return status_e::non_standard_encoding;
213 }
214
215 // if next character is not part of the double character we add this as code point
216 if (!is_low_surrogate(next_character)) {
217 if (!comply_with_standard) {
218 code_points.push_back(this_character);
219 continue;
220 }
221 return status_e::non_standard_encoding;
222 }
223
224 // it turned out we have a double character
225 was_double_character = true;
226
227 // do decoding "double UTF-16" -> UTF-32:
228 // https://en.wikipedia.org/wiki/UTF-16#Code_points_from_U+010000_to_U+10FFFF
229 char32_t high_code_point = (this_character - constants::high_surrogate_start) << 10;
230 char16_t low_code_point = next_character - constants::low_surrogate_start;
231 char32_t code_point = high_code_point + low_code_point + constants::supplementary_plane_offset;
232 code_points.push_back(code_point);
233
234 continue;
235 }
236
237 // if not a double character add this as next code point
238 code_points.push_back(this_character);
239 }
240 // return resulting string
241 utf32_s = code_points;
242 return status_e::success;
243}
