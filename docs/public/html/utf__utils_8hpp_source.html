310status_e utf::conversion::utf8_to_utf32(const std::basic_string_view<char8_t>& utf8_sv, std::basic_string<char32_t>& utf32_s, bool comply_with_standard = false) {
311 std::basic_string<char32_t> code_points;
312
313 size_t code_unit_count = utf8_sv.size();
314 uint8_t bits_left = 0;
315 uint8_t bytes_left = 0;
316 for (size_t index = 0; index < code_unit_count; index++) {
317 char8_t this_code_unit = utf8_sv[index];
318 // if first bit is zero it's ANSI
319 if (this_code_unit >> 7 == 0) {
320 code_points.push_back(this_code_unit);
321
322 continue;
323 }
324 // if first byte is "is trailing" for some reason
325 if (this_code_unit >> 6 == trailing_byte_marker) {
326 return status_e::trailing_without_leading;
327 }
328 // if first byte denotes double character
329 if (this_code_unit >> 5 == double_byte_marker) {
330 // get rid of markers
331 uint16_t first_bits = static_cast<uint16_t>(this_code_unit << 6);
332
333 // next byte
334 if (++index >= code_unit_count) {
335 return status_e::character_cut_off;
336 }
337 // ~next byte
338
339 uint16_t last_bits = static_cast<uint16_t>((utf8_sv[index] << 2) >> 2);
340 // create code point
341 char16_t code_point = first_bits + last_bits;
342 code_points.push_back(code_point);
343
344 continue;
345 }
346 // if first byte denotes triple character
347 if (this_code_unit >> 4 == triple_byte_marker) {
348 // get rid of markers
349 uint16_t first_bits = static_cast<uint16_t>(this_code_unit) << 12;
350
351 // next byte
352 if(++index >= code_unit_count) {
353 return status_e::character_cut_off;
354 }
355 // ~next byte
356 uint16_t middle_bits = static_cast<uint16_t>((utf8_sv[index] << 2) >> 2) << 6;
357
358 // next byte
359 if(++index >= code_unit_count) {
360 return status_e::character_cut_off;
361 }
362 // ~next byte
363 uint16_t last_bits = static_cast<uint16_t>((utf8_sv[index] << 2) >> 2);
364
365 char16_t code_point = first_bits + middle_bits + last_bits;
366 code_points.push_back(code_point);
367 continue;
368 }
369 // if first byte denotes quadruple character
370 if(this_code_unit >> 3 == quadruple_byte_marker) {
371 uint32_t first_bits = this_code_unit << 18;
372
373 // next byte
374 if(++index >= code_unit_count) {
375 return status_e::character_cut_off;
376 }
377 // ~next byte
378 uint32_t second_bits = static_cast<uint32_t>((utf8_sv[index] << 2) >> 2) << 12;
379
380 // next byte
381 if(++index >= code_unit_count) {
382 return status_e::character_cut_off;
383 }
384 // ~next byte
385 uint32_t third_bits = static_cast<uint32_t>((utf8_sv[index] << 2) >> 2) << 6;
386
387 // next byte
388 if(++index >= code_unit_count) {
389 return status_e::character_cut_off;
390 }
391 // ~next byte
392 uint32_t last_bits = static_cast<uint32_t>((utf8_sv[index] << 2) >> 2);
393
394
395 char32_t code_point = first_bits + second_bits + third_bits + last_bits;
396 code_points.push_back(code_point);
397 continue;
398 }
399 }
400
401
402
403
404 return status_e::success;
405}

407status_e utf::conversion::utf16_to_utf8(const std::basic_string_view<char16_t>& utf16_sv, std::basic_string<char8_t>& utf8_s, bool comply_with_standard = false) {
408 // Convert to UTF-32
409 std::basic_string<char32_t> code_points;
410 status_e status = utf16_to_utf32(utf16_sv, code_points, comply_with_standard);
411 if (status < status_e::success) {
412 return status;
413 }
414 // Convert to UTF-8
415 std::basic_string<char8_t> result_string;
416 status = utf32_to_utf8(code_points, result_string, comply_with_standard);
417 if (status < status_e::success) {
418 return status;
419 }
420
421 utf8_s = result_string;
422 return status_e::success;

425status_e utf::conversion::utf16_to_utf32(const std::basic_string_view<char16_t>& utf16_sv, std::basic_string<char32_t>& utf32_s, bool comply_with_standard = false) {
426 // vector of UTF-32 characters (code points)
427 std::basic_string<char32_t> code_points;
428
429 bool was_double_character = false;
430 for (auto character_it = utf16_sv.begin(); character_it < utf16_sv.end(); character_it++) {
431 // skip this character, because we handled it last iteration
432 if (was_double_character) {
433 was_double_character = false;
434 continue;
435 }
436 // get this character
437 char16_t this_character = *character_it;
438
439 // check if this character is the last one
440 const bool exists_next_character = (character_it + 1) != utf16_sv.end();
441 char16_t next_character;
442
443 // get next character if there is one
444 if (exists_next_character) {
445 next_character = *(character_it + 1);
446 }
447
448 // if can be part of double character
449 if (is_high_surrogate(this_character)) {
450 // if there is no next character we add this as code point
451 if (!exists_next_character) {
452 if (!comply_with_standard) {
453 code_points.push_back(this_character);
454 continue;
455 }
456 return status_e::non_standard_encoding;
457 }
458
459 // if next character is not part of the double character we add this as code point
460 if (!is_low_surrogate(next_character)) {
461 if (!comply_with_standard) {
462 code_points.push_back(this_character);
463 continue;
464 }
465 return status_e::non_standard_encoding;
466 }
467
468 // it turned out we have a double character
469 was_double_character = true;
470
471 // do decoding "double UTF-16" -> UTF-32:
472 // https://en.wikipedia.org/wiki/UTF-16#Code_points_from_U+010000_to_U+10FFFF
473 char32_t high_code_point = (this_character - high_surrogate_start) << 10;
474 char16_t low_code_point = next_character - low_surrogate_start;
475 char32_t code_point = high_code_point + low_code_point + supplementary_plane_offset;
476 code_points.push_back(code_point);
477
478 continue;
479 }
480
481 // if not a double character add this as next code point
482 code_points.push_back(this_character);
483 }
484 // return resulting string
485 utf32_s = code_points;
486 return status_e::success;
487}

489status_e utf::conversion::utf32_to_utf8(const std::basic_string_view<char32_t>& utf32_sv, std::basic_string<char8_t>& utf8_s, bool comply_with_standard = false) {
490 std::basic_string<char8_t> result;
491
492 for (const char32_t this_code_point : utf32_sv) {
493 if (this_code_point > four_byte_boundary) {
494 return status_e::undefined_error;
495 }
496 if (this_code_point <= one_byte_boundary) {
497 result.push_back(static_cast<char8_t>(this_code_point));
498 continue;
499 }
500 if (this_code_point <= two_byte_boundary) {
501 uint16_t cp_16 = static_cast<uint16_t>(this_code_point);
502
503 uint8_t first_5_bits = cp_16 >> 6;
504 uint8_t last_6_bits = (cp_16 << 10) >> 10;
505
506 result.push_back((double_byte_marker << 5) + first_5_bits);
507 result.push_back((trailing_byte_marker << 6) + last_6_bits);
508
509 continue;
510 }
511 if (this_code_point <= three_byte_boundary) {
512 uint16_t cp_16 = static_cast<uint16_t>(this_code_point);
513
514 bool wrong_encoding = comply_with_standard &&
515 cp_16 < low_surrogate_start &&
516 cp_16 > high_surrogate_start;
517 if (wrong_encoding) {
518 return status_e::non_standard_encoding;
519 }
520
521 uint8_t first_4_bits = cp_16 >> 12;
522 uint8_t middle_6_bits = (cp_16 << 4) >> 10;
523 uint8_t last_6_bits = (cp_16 << 10) >> 10;
524
525 result.push_back((triple_byte_marker << 4) + first_4_bits);
526 result.push_back((trailing_byte_marker << 6) + middle_6_bits);
527 result.push_back((trailing_byte_marker << 6) + last_6_bits);
528
529 continue;
530 }
531 if (this_code_point <= four_byte_boundary) {
532 uint8_t first_3_bits = this_code_point >> 18;
533 uint8_t second_6_bits = (this_code_point << 14) >> 26;
534 uint8_t third_6_bits = (this_code_point << 20) >> 26;
535 uint8_t last_6_bits = (this_code_point << 26) >> 26;
536
537 result.push_back((quadruple_byte_marker << 5) + first_3_bits);
538 result.push_back((trailing_byte_marker << 6) + second_6_bits);
539 result.push_back((trailing_byte_marker << 6) + third_6_bits);
540 result.push_back((trailing_byte_marker << 6) + last_6_bits);
541
542 continue;
543 }
544 }
545
546 return status_e::success;
547}
