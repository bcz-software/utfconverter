303status_e utf::conversion::utf32_to_utf8(const std::basic_string_view<char32_t>& utf32_sv, std::basic_string<char8_t>& utf8_s, bool comply_with_standard = false) {
304 std::basic_string<char8_t> result;
305
306 for (auto it = utf32_sv.begin(); it < utf32_sv.end(); it++) {
307 char32_t this_code_point = *it;
308 if (this_code_point > four_byte_boundary) {
309 return status_e::undefined_error;
310 }
311 if (this_code_point <= one_byte_boundary) {
312 result.push_back(this_code_point);
313 continue;
314 }
315 if (this_code_point <= two_byte_boundary) {
316 uint16_t cp_16 = static_cast<uint16_t>(this_code_point);
317
318 uint8_t first_5_bits = cp_16 >> 6;
319 uint8_t last_6_bits = (cp_16 << 10) >> 10;
320
321 result.push_back((double_byte_marker << 5) + first_5_bits);
322 result.push_back((trailing_byte_marker << 6) + last_6_bits);
323
324 continue;
325 }
326 if (this_code_point <= three_byte_boundary) {
327 uint16_t cp_16 = static_cast<uint16_t>(this_code_point);
328
329 bool wrong_encoding = comply_with_standard &&
330 cp_16 < low_surrogate_start &&
331 cp_16 > high_surrogate_start;
332 if (wrong_encoding) {
333 return status_e::non_standard_encoding;
334 }
335
336 uint8_t first_4_bits = cp_16 >> 12;
337 uint8_t middle_6_bits = (cp_16 << 4) >> 10;
338 uint8_t last_6_bits = (cp_16 << 10) >> 10;
339
340 result.push_back((triple_byte_marker << 4) + first_4_bits);
341 result.push_back((trailing_byte_marker << 6) + middle_6_bits);
342 result.push_back((trailing_byte_marker << 6) + last_6_bits);
343
344 continue;
345 }
346 if (this_code_point <= four_byte_boundary) {
347 uint8_t first_3_bits = this_code_point >> 18;
348 uint8_t second_6_bits = (this_code_point << 14) >> 26;
349 uint8_t third_6_bits = (this_code_point << 20) >> 26;
350 uint8_t last_6_bits = (this_code_point << 26) >> 26;
351
352 result.push_back((quadruple_byte_marker << 5) + first_3_bits);
353 result.push_back((trailing_byte_marker << 6) + second_6_bits);
354 result.push_back((trailing_byte_marker << 6) + third_6_bits);
355 result.push_back((trailing_byte_marker << 6) + last_6_bits);
356
357 continue;
358 }
359 }
360
361 return status_e::success;
362}

364status_e utf::conversion::utf16_to_utf32(const std::basic_string_view<char16_t>& utf16_sv, std::basic_string<char32_t>& utf32_s, bool comply_with_standard = false) {
365 // vector of UTF-32 characters (code points)
366 std::basic_string<char32_t> code_points;
367
368 bool was_double_character = false;
369 for (auto character_it = utf16_sv.begin(); character_it < utf16_sv.end(); character_it++) {
370 // skip this character, because we handled it last iteration
371 if (was_double_character) {
372 was_double_character = false;
373 continue;
374 }
375 // get this character
376 char16_t this_character = *character_it;
377
378 // check if this character is the last one
379 const bool exists_next_character = (character_it + 1) != utf16_sv.end();
380 char16_t next_character;
381
382 // get next character if there is one
383 if (exists_next_character) {
384 next_character = *(character_it + 1);
385 }
386
387 // if can be part of double character
388 if (is_high_surrogate(this_character)) {
389 // if there is no next character we add this as code point
390 if (!exists_next_character) {
391 if (!comply_with_standard) {
392 code_points.push_back(this_character);
393 continue;
394 }
395 return status_e::non_standard_encoding;
396 }
397
398 // if next character is not part of the double character we add this as code point
399 if (!is_low_surrogate(next_character)) {
400 if (!comply_with_standard) {
401 code_points.push_back(this_character);
402 continue;
403 }
404 return status_e::non_standard_encoding;
405 }
406
407 // it turned out we have a double character
408 was_double_character = true;
409
410 // do decoding "double UTF-16" -> UTF-32:
411 // https://en.wikipedia.org/wiki/UTF-16#Code_points_from_U+010000_to_U+10FFFF
412 char32_t high_code_point = (this_character - high_surrogate_start) << 10;
413 char16_t low_code_point = next_character - low_surrogate_start;
414 char32_t code_point = high_code_point + low_code_point + supplementary_plane_offset;
415 code_points.push_back(code_point);
416
417 continue;
418 }
419
420 // if not a double character add this as next code point
421 code_points.push_back(this_character);
422 }
423 // return resulting string
424 utf32_s = code_points;
425 return status_e::success;
426}
